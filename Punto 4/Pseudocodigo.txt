Proceso Punto_4
	Definir n,cuadrado,cubo,doble,triple Como Real;
	Escribir 'Ingrse un numero';

	cuadrado <- (n*n);
	cubo <- (n*n*n);
	doble <- (n+n);
	triple <- (n+n+n);
	Escribir 'El cuadrado es: ',cuadrado;
	Escribir 'El cubo es: ',cubo;
	Escribir 'El doble es: ',doble;
	Escribir 'El triple es: ',triple;
